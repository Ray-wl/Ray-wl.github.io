axios实现跨域访问

安装axios，并在main.js中引入

import Axios from 'axios

// 把axios挂载到Vue原型中，这样在vue的每个组件中都可以通过使用this.$axios发起axios请求，不用再每个组件中都再次import Axios from 'axios'
Vue.prototype.$axios = Axios

// 重点就在这里，Vue.prototype.HOME = '/api' 是一个定值，默认指向localhost，如本文所修改的，将'/api'挂载到变量HOME上，项目中需要访问'/api'直接使用this.HOME就可以（'/api'指向的路径要看config/index.js文件中target的具体设置，见下文）。
Vue.prototype.HOME = '/api'

配置config/index.js文件

module.exports = {
  dev: {
    // Paths
    assetsSubDirectory: 'static',
    assetsPublicPath: '/',
    proxyTable: {                         // axios跨域处理
      '/api': {                           // 此处并非和url一致
        target:'https：//3g.163.com',    // 这里是需要跨域的链接
        changeOrigin:true,                 // 允许跨域
        pathRewrite:{
          '^/api': ''                    // 这里设置的目的就是将target里面的地址用'/api'来代替，后面在代码里写'/api'就等于写了target里的这个地址，比如我在项目中需要访问'https://3g.163.com/touch/reconstruct/article/list/BBM54PGAwangning/0-10.html'，我可以写成'/api/touch/reconstruct/article/list/BBM54PGAwangning/0-10.html'或者this.HOME+'/touch/reconstruct/article/list/BBM54PGAwangning/0-10.html',都可以访问到相应的接口
        }
      }
    }
  }
}

在跨域文件中使用

created () {
    var url = this.HOME + '/touch/reconstruct/article/list/BBM54PGAwangning/0-10.html'
    // 或者可以使用 var url = '/api/touch/reconstruct/article/list/BBM54PGAwangning/0-10.html'，这两种都可以实现访问
    this.$axios({
        method: 'get'
        url: url,
        data: {}        // 这里配置data
    }).then(res => {
        console.log(res)
    }).catch(err => {
        console.log(err)    //访问出现问题这里抛出错误
    })
}

通过以上设置就可以完成跨域访问

重点重点重点：重要的事情要说三遍

在Vue项目中如果我们修改了config里面的文件，请千万要“重新启动项目，重新启动项目，重新启动项目”，不然一定会报错。

以上都是自己实现小项目时实现的跨域访问，引文本人的技术仅限前端，对于后端实现的跨域无法实现，至于实际工作中的项目，或其他重要线上项目后端实现跨域才是王道。
